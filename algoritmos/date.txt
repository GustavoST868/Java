
package main;
  //libraries
  import javax.swing.JOptionPane;

public class Main {
    
 
    //Date class
    public static class Date{
        
        
            //private attributes
            public int month;
            public int day;
            public int year;
    
    
             //constructor
            public  Date(int month,int day,int year){
                this.month = month;
                this.day   = day;
                this.year  = year;
            }
            
            
            //return month
            public int getMonth(){
                return month;
            }
            
            
            //function to set the month
            public void setMonth(int month){
                this.month = month;
            }
            
            
            //returt day
            public int getDay(){
                return day;
            }
            
            
            
            //function to set the day
            public void setDay(int day){
                this.day = day;
            }
            
            
            //retun year
            public int getYear(){
                return year;
            }
            
            
            //function to set the year
            public void setYear(int year){
                this.year = year;
            }
            
            
            //function to print the Date
            public void printDate(){
                JOptionPane.showMessageDialog(null,"["+day+"/"+month+"/"+year+"]");
            }
    }
 
    
    
    //class to test  functions
    public static class dateTest{

        JOptionPane.showMessageDialog(null,"Day: "   + date.getDay());
        JOptionPane.showMessageDialog(null,"Year: "  + date.getYear());
        
        
        //use the method to showing the date
        date.printDate();

        }
         
        
    }
    
    
    public static void main(String[] args) {
